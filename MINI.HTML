function showModel() { // إخفاء جميع أقسام النماذج document.getElementById('AModel').classList.add('hidden'); document.getElementById('BModel').classList.add('hidden'); document.getElementById('CModel').classList.add('hidden'); // إخفاء جميع divs الخاصة بالبيانات var allContentDivs = document.querySelectorAll('.content-div'); allContentDivs.forEach(function(div) { div.classList.add('hidden'); // إخفاء كل divs الموجودة }); // إظهار القسم المناسب بناءً على نوع السيارة المختار var car = document.getElementById('carSelect').value; if (car) { currentModelId = car + 'Model'; // حفظ الموديل الحالي document.getElementById(currentModelId).classList.remove('hidden'); document.querySelector('.selected-info').style.display = 'block'; // إظهار selected-info عند اختيار نموذج document.getElementById('backButton').style.display = 'none'; // إخفاء زر العودة } } function goToDiv(divId, modelId) { // إخفاء جميع divs var allContentDivs = document.querySelectorAll('.content-div'); allContentDivs.forEach(function(div) { div.classList.add('hidden'); // إخفاء كل divs الموجودة }); // إظهار الـ div المطلوب document.getElementById(divId).classList.remove('hidden'); // إخفاء الأزرار الأخرى document.getElementById(modelId).classList.add('hidden'); document.getElementById('backButton').style.display = 'block'; // إظهار زر العودة } function goBack() { // إخفاء جميع divs المتعلقة بالنماذج document.querySelectorAll('.content-div').forEach(function(div) { div.classList.add('hidden'); }); // إعادة عرض الموديل الحالي فقط if (currentModelId) { document.getElementById(currentModelId).classList.remove('hidden'); } // إخفاء زر العودة document.getElementById('backButton').style.display = 'none'; }
